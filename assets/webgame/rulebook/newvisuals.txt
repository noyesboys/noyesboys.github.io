// ================================================================
// THE NEW BUD GENERATOR HELPER
// ================================================================
function getBudSvg(cx, cy, scale, rot, budColor, budDark, isCrispy, isMoldy, pistilColor) {
  const nugPath = "M 0,-16 C 8,-8 12,-2 10,4 C 8,10 12,14 6,16 C 0,18 -6,16 -10,12 C -14,6 -10,-2 -6,-8 C -8,-12 0,-16 0,-16 Z";
  const texture = "M -4,2 Q 0,6 4,2 M -2,-4 Q 2,-2 6,-6 M -6,-8 Q -2,-4 0,-10";
  const leaves = "M -8,-6 L -16,-12 L -10,-2 Z M 8,-4 L 18,-8 L 10,2 Z M 0,16 L -4,24 L -8,14 Z M -10,6 L -18,8 L -12,12 Z";
  const pistils = "M -6,8 Q -12,14 -16,10 M 6,10 Q 14,16 16,8 M 0,-16 Q 4,-24 8,-20 M -8,-2 Q -16,-6 -14,-10 M 8,2 Q 16,-2 14,2";

  const mold = isMoldy ? `
    <circle cx="0" cy="0" r="10" fill="#e2e8f0" opacity="0.45" filter="blur(2px)"/>
    <circle cx="-3" cy="5" r="4" fill="#94a3b8" opacity="0.6"/>
    <circle cx="4" cy="-4" r="3" fill="#cbd5e1" opacity="0.7"/>
  ` : '';
  
  const crispyLines = isCrispy ? `<path d="M -5,-5 L 5,5 M -5,5 L 5,-5 M 0,8 L 0,12" stroke="${budDark}" stroke-width="0.8" opacity="0.6"/>` : '';

  return `
  <g transform="translate(${cx}, ${cy}) scale(${scale}) rotate(${rot})">
      <path d="${leaves}" fill="${budColor}" filter="brightness(1.1)"/>
      <path d="${nugPath}" fill="${budColor}"/>
      <path d="${texture}" stroke="${budDark}" stroke-width="1.2" fill="none" opacity="0.6" stroke-linecap="round"/>
      <path d="M 0,-5 Q 4,2 2,8 Q 0,10 -2,8 Q -4,2 0,-5 Z" fill="${budDark}" opacity="0.35"/>
      <path d="${pistils}" stroke="${pistilColor || '#d97706'}" stroke-width="1.5" fill="none" stroke-linecap="round"/>
      ${crispyLines}
      ${mold}
  </g>`;
}

// ================================================================
// UPDATED DRYING SPRITE
// ================================================================
function getDryingSprite(batch, drMock) {
  const rh   = batch.dryRH != null ? batch.dryRH : 62;
  const q    = batch.quality || 100;
  const fanOn = drMock.fanOn;
  const fanSpeed = fanOn ? drMock.fanSpeed : 0;

  const isMoldRisk   = rh > 75;
  const isTooWet     = rh > 68 && rh <= 75;
  const isGoodDrying = rh >= 62 && rh <= 68;
  const isSweetSpot  = rh >= 58 && rh < 62;
  const isOverDried  = rh < 40;

  const budColor   = isOverDried  ? '#7a5c2e' : isMoldRisk   ? '#4a7a2e' : isGoodDrying ? '#5db356' : isSweetSpot  ? '#8bc34a' : '#6ab04c';
  const budDark    = isOverDried  ? '#5c4020' : '#2d5a1b';
  const pistilCol  = isOverDried  ? '#78350f' : isTooWet ? '#ea580c' : '#d97706'; 
  const stemColor  = isOverDried  ? '#8d6c3a' : '#5a8a3a';
  const rhColor    = isMoldRisk   ? '#ef4444' : isTooWet ? '#f97316' : isGoodDrying ? '#22d3ee' : isSweetSpot ? '#4ade80' : isOverDried ? '#d97706' : '#a3e635';
  
  const swayAnim   = fanSpeed > 60 ? 'swayFast 1.2s ease-in-out infinite' : fanSpeed > 0 ? 'sway 2.2s ease-in-out infinite' : 'sway 5s ease-in-out infinite';
  const moldy      = isMoldRisk && (q < 80);
  const crispy     = isOverDried;

  const branch = (x, delay, scale) => {
    const sy = scale || 1;
    return `
    <g style="animation:${swayAnim};animation-delay:${delay}s;transform-origin:${x}px 20px">
      <line x1="${x}" y1="8" x2="${x}" y2="42" stroke="#8d6c3a" stroke-width="1.5" stroke-dasharray="${crispy?'3,2':'none'}"/>
      <line x1="${x}" y1="42" x2="${x}" y2="${42+65*sy}" stroke="${stemColor}" stroke-width="${3*sy}" stroke-linecap="round"/>
      <line x1="${x}" y1="${52}" x2="${x-18*sy}" y2="${72}" stroke="${stemColor}" stroke-width="${2*sy}"/>
      <line x1="${x}" y1="${58}" x2="${x+18*sy}" y2="${76}" stroke="${stemColor}" stroke-width="${2*sy}"/>
      <line x1="${x}" y1="${68}" x2="${x-14*sy}" y2="${88}" stroke="${stemColor}" stroke-width="${2*sy}"/>
      <line x1="${x}" y1="${74}" x2="${x+14*sy}" y2="${92}" stroke="${stemColor}" stroke-width="${2*sy}"/>
      ${getBudSvg(x-20*sy, 74, sy*0.8, 145, budColor, budDark, crispy, moldy, pistilCol)}
      ${getBudSvg(x+20*sy, 78, sy*0.8, -145, budColor, budDark, crispy, moldy, pistilCol)}
      ${getBudSvg(x-16*sy, 90, sy*0.9, 160, budColor, budDark, crispy, moldy, pistilCol)}
      ${getBudSvg(x+16*sy, 94, sy*0.9, -160, budColor, budDark, crispy, moldy, pistilCol)}
      ${getBudSvg(x, 108*sy, sy*1.4, 180, budColor, budDark, crispy, moldy, pistilCol)}
    </g>`;
  };

  return `<svg viewBox="0 0 300 200" xmlns="http://www.w3.org/2000/svg" style="width:100%;height:100%">
    <rect width="300" height="200" fill="#060e06"/>
    <line x1="20" y1="10" x2="280" y2="10" stroke="#5a4030" stroke-width="2"/>
    ${branch(62, 0, 0.9)}
    ${branch(150, 0.4, 1.0)}
    ${branch(238, 0.8, 0.95)}
    <rect x="8" y="8" width="54" height="20" rx="4" fill="rgba(0,0,0,.7)" stroke="${rhColor}" stroke-width=".8"/>
    <text x="35" y="22" text-anchor="middle" font-family="monospace" font-size="11" font-weight="bold" fill="${rhColor}">${rh.toFixed(1)}%</text>
  </svg>`;
}

// ================================================================
// UPDATED CURING JAR SPRITE - WITH COLOR SHIFT
// ================================================================
function getCuringSprite(batch, hrsSince) {
  const rh         = batch.jarRH != null ? batch.jarRH : 62;
  const q          = batch.quality || 100;
  const burps      = batch.burpsDone || 0; // The color driver
  
  const needsBurp  = rh > 65 && hrsSince >= 4;
  const overdue    = rh > 70 || (needsBurp && hrsSince >= 10);
  const belowFifty = rh < 50;
  const sweetSpot  = rh >= 54 && rh <= 64;

  // COLOR SHIFT LOGIC (The "Red/Dark to Green" Cure)
  // Fresh weed (low burps) = Darker, deeper green (higher chlorophyll)
  // Curing weed (mid burps) = Transitions to standard green
  // Fully cured (high burps) = Vibrant, lighter green
  
  let budColor = '#4a7a2e'; // Default "Fresh"
  let pistilCol = '#ea580c'; // Default "Fresh" hairs
  
  if (belowFifty) {
    budColor = '#7a5c2e'; 
    pistilCol = '#78350f';
  } else {
    // Smooth transition based on burps (up to 6)
    if (burps >= 6) {
       budColor = '#8bc34a'; // Cured Light
       pistilCol = '#fbbf24'; // Golden hairs
    } else if (burps >= 3) {
       budColor = '#6ab04c'; // Mid-Cure
       pistilCol = '#d97706'; // Orange hairs
    } else {
       budColor = '#2d5a1b'; // Very Fresh/Dark
       pistilCol = '#c2410c'; // Deep Red hairs
    }
  }

  const budDark = belowFifty ? '#5c4020' : '#1b3a11';
  const jarStroke = overdue ? '#ef4444' : needsBurp ? '#fbbf24' : sweetSpot ? '#4ade80' : belowFifty ? '#d97706' : '#22d3ee';
  const jarFill = overdue ? 'rgba(239,68,68,.08)' : needsBurp ? 'rgba(251,191,36,.08)' : sweetSpot ? 'rgba(74,222,128,.08)' : belowFifty ? 'rgba(217,119,6,.06)' : 'rgba(34,211,238,.06)';
  const glowAnim = overdue ? 'jarDanger 1.5s ease-in-out infinite' : needsBurp ? 'jarWarn 2s ease-in-out infinite' : sweetSpot ? 'jarGlow 3s ease-in-out infinite' : 'none';

  return `<svg viewBox="0 0 300 200" xmlns="http://www.w3.org/2000/svg" style="width:100%;height:100%">
    <rect width="300" height="200" fill="#080c12"/>
    <g style="animation:${glowAnim}">
      <rect x="102" y="62" width="96" height="124" rx="8" fill="${jarFill}" stroke="${jarStroke}" stroke-width="1.5"/>
      <g clip-path="url(#jarClip)">
        ${getBudSvg(120, 175, 1.4, -40, budColor, budDark, belowFifty, false, pistilCol)}
        ${getBudSvg(150, 178, 1.5,  15, budColor, budDark, belowFifty, false, pistilCol)}
        ${getBudSvg(180, 172, 1.3,  60, budColor, budDark, belowFifty, false, pistilCol)}
        ${getBudSvg(132, 155, 1.3, -20, budColor, budDark, belowFifty, false, pistilCol)}
        ${getBudSvg(168, 150, 1.2,  30, budColor, budDark, belowFifty, false, pistilCol)}
        ${getBudSvg(150, 130, 1.3,  -5, budColor, budDark, belowFifty, false, pistilCol)}
        ${getBudSvg(112, 140, 1.0, -80, budColor, budDark, belowFifty, false, pistilCol)}
        ${getBudSvg(188, 135, 1.0,  85, budColor, budDark, belowFifty, false, pistilCol)}
        ${getBudSvg(150, 105, 1.1,  15, budColor, budDark, belowFifty, false, pistilCol)}
      </g>
      <rect x="98" y="56" width="104" height="16" rx="4" fill="${jarFill}" stroke="${jarStroke}" stroke-width="1.2"/>
      <rect x="94" y="48" width="112" height="14" rx="4" fill="rgba(0,0,0,.7)" stroke="${jarStroke}" stroke-width="1"/>
    </g>

    <!-- BURP INDICATOR -->
    <rect x="110" y="70" width="80" height="6" rx="3" fill="#000" opacity="0.4"/>
    <rect x="110" y="70" width="${Math.min(burps*8, 80)}" height="6" rx="3" fill="${budColor}"/>

    <rect x="8" y="8" width="54" height="20" rx="4" fill="rgba(0,0,0,.75)" stroke="${jarStroke}" stroke-width=".8"/>
    <text x="35" y="22" text-anchor="middle" font-family="monospace" font-size="11" font-weight="bold" fill="${jarStroke}">${rh.toFixed(1)}%</text>
  </svg>`;
}

function updateUI() {
  const rh = parseFloat(document.getElementById('inp-rh').value);
  const q = parseFloat(document.getElementById('inp-q').value);
  const fan = parseFloat(document.getElementById('inp-fan').value);
  const fanOn = document.getElementById('inp-fan-on').checked;
  const burps = parseInt(document.getElementById('inp-burps').value);
  const hrs = parseFloat(document.getElementById('inp-hrs').value);

  document.getElementById('val-rh').innerText = rh + '%';
  document.getElementById('val-q').innerText = q + '%';
  document.getElementById('val-fan').innerText = fan + '%';
  document.getElementById('val-burps').innerText = burps;
  document.getElementById('val-hrs').innerText = hrs + 'h';

  const batchMock = { dryRH: rh, jarRH: rh, quality: q, burpsDone: burps };
  const drMock = { fanOn: fanOn, fanSpeed: fan };

  document.getElementById('render-drying').innerHTML = getDryingSprite(batchMock, drMock);
  document.getElementById('render-curing').innerHTML = getCuringSprite(batchMock, hrs);
}
updateUI();
